# If reprository made in Localhost then-->
1) git init
2) git status
3) git add .
4) git commit -m "Comment"
5) git remote add origin [HTTPS github link]
6) git remote -v
7) git branch -M [renameBranch according to github]
8) git push -u origin main

if push is not working->
git pull --rebase origin [branchName]
8) git push -u origin main
